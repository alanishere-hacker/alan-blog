二叉树遍历与应用：表达式二叉树的构建与求值

一、实验目的及要求
实验目的：
（1）掌握二叉树的二叉链表存储结构。
（2）验证二叉树遍历算法及其他基本操作的实现。
（3）理解算法与程序的关系，能够将二叉树的算法转换为对应的程序。
（4）学会使用二叉树来解决各种实际问题，掌握利用二叉树求解表达式的值。
实验要求：
(1) 程序要添加适当的注释，程序的书写要采用缩进格式。
(2) 程序要具在一定的健壮性，即当输入数据非法时，程序也能适当地做出反应，如插入删除时指定的位置不对等等。
(3) 程序要做到界面友好，在程序运行时用户可以根据相应的提示信息进行操作。
(4) 根据实验报告模板详细书写实验报告。
(5) 以班级为单位提交实验报告，文件命名：学号姓名实验一实验报告，学号姓名实验一代码。
二、实验原理与内容
（1）定义二叉树的结点结构。
（2）创建表达式二叉树。
（3）先序遍历表达式二叉树得到表达式的前缀表示。
（4）中序遍历表达式二叉树得到表达式的中缀表示。
（5）后序遍历表达式二叉树得到表达式的后缀表示。
（6）计算表达式的值。

//二叉树的应用-表达式求值
#include<iostream>
#include<fstream>
#include<math.h>
#include<string.h>
using namespace std;
#define MAXSIZE 100//表达式字符串最大值
#define N 10//表达式项的最大长度
char expression[MAXSIZE ]; //存储表达式字符串
char str[MAXSIZE][N];//存储表达式操作数、运算符
int expstart,expend;//表达式开始、结束位置
typedef char ElemType;
typedef struct BiTNode
{
ElemType data[N];
BiTNode* lchild,* rchild;
}BiTNode,*BiTree;
bool Judge(char (*str)[N],int expstart,int expend)//判断当前表达式是否需要消除最外层的括号
{
bool result=false;
if(strcmp(str[expstart],"(")==0&&strcmp(str[expend],")")==0)
{
char flag=')';
for(int i=expstart+1; i<expend; i++)
{
if(strcmp(str[i],"(")==0||strcmp(str[i],")")==0)
flag=str[i][0];
}
if(flag==')')
result=true;
}
return result;
}
int GetOprt(char (*str)[N],int expstart,int expend)
//取得当前表达式中优先级最低的操作符在表达式expression中的位置
{
int position=-1;
int sign=0;
int t=1;
while(position==-1)
{
for(int i=expstart; i<=expend; i++)
{
if(strcmp(str[i],"(")==0)
{
i++;
sign=1; //有一个左括号
while(sign>0)
{
//sign=0，表示括号去完了
if(strcmp(str[i],"(")==0)
sign++;
if(strcmp(str[i],")")==0)
sign--;
i++;
}
i--;
}
if(t==1)
{
if(strcmp(str[i],"+")==0||strcmp(str[i],"-")==0)
position=i; //记录括号外优先级最低的加减符号位置
}
else
{
if(strcmp(str[i],"*")==0||strcmp(str[i],"/")==0)
position=i; //记录括号外优先级最低的乘除符号位置
}
}
t++;
}
return position;
}
//递归建立二叉树,根据表达式运算符的优先级，从低到高以先序递归的方式构建表达式二叉树
BiTNode *CreateBT(char (*str)[N], int expstart, int expend)
{
int position=-1;
BiTNode *BT;
if(expend-expstart==0)
{
BT=(BiTNode*)malloc(sizeof(BiTNode));
strcpy(BT->data,str[expstart]);
BT->lchild=NULL;
BT->rchild=NULL;
return BT;
}
position=GetOprt(str,expstart,expend);
BT=(BiTNode*)malloc(sizeof(BiTNode));
strcpy(BT->data,str[position]);
bool removable=Judge(str,expstart,position-1);
if(removable)
BT->lchild=CreateBT(str,expstart+1,position-2);
else
BT->lchild=CreateBT(str,expstart,position-1);
removable=Judge(str,position+1,expend);
if(removable)
BT->rchild=CreateBT(str,position+2,expend-1);
else
BT->rchild=CreateBT(str,position+1,expend);
return BT;
}
//将叶节点中以字符形式存储的数字转换成浮点型的
double StrToDigital(char str[])
{
int i,len,position;
double sum;
len=strlen(str);
for(i=0; i<len; i++)
if(str[i]=='.')
break;
position=i;
i=0;
sum=0;
while(i<len)
{
if(i<position)
sum+=(str[i]-'0')*pow(10,position-i-1);
else if(i>position)
sum+=(str[i]-'0')*pow(10,position-i);
i++;
}
return sum;
}
//先序遍历
void PreOrderBT(BiTree BT)
{
if(BT)
{
cout<<BT->data;
PreOrderBT(BT->lchild);
PreOrderBT(BT->rchild);
}
}
//中序遍历
void InOrderBT(BiTree BT)
{
if(BT)
{
InOrderBT(BT->lchild);
cout<<BT->data;
InOrderBT(BT->rchild);
}
}
//
//后序遍历
void PostOrderBT(BiTree BT)
{
if(BT)
{
PostOrderBT(BT->lchild);
PostOrderBT(BT->rchild);
cout<<BT->data;
}
}
//求整个表达式的结果
double CalulateValue(BiTNode* BT)
{
double result=0;
if(BT->lchild==NULL&&BT->rchild==NULL)
{
result=StrToDigital(BT->data);
return result;
}
double Lresult=CalulateValue(BT->lchild);
double Rresult=CalulateValue(BT->rchild);
switch(BT->data[0])
{
case '+':
return Lresult+Rresult;
break;
case '-':
return Lresult-Rresult;
break;
case '*':
return Lresult*Rresult;
break;
case '/':
return Lresult/Rresult;
break;
}
return result;
}
//初始化，输入表达式expression并将每项操作数或运算符存入字符数组str中
void Init()
{
int i,j,k;
cout<<"输入需要求值的表达式(如：23-6*(5-2)+4/3 英文状态输入):\n";
gets(expression);
int length=strlen(expression);
i=0;
k=0;
while(i<length) //将表达式存储到str数组中
{
if(expression[i]=='+'||expression[i]=='-'||expression[i]=='*'||expression[i]=='/'||expression[i]=='('||expression[i]==')')
str[k++][0]=expression[i++]; //存储操作符
else //存储数字
{
j=0;
while((expression[i]>='0'&&expression[i]<='9') || expression[i]=='.')
{
str[k][j]=expression[i];
i++;
j++;
}
k++;
}
}
expstart=0;
expend=k-1;
};
//主函数
int main()
{
char c;
bool removable;
do
{
double result;
BiTNode *BT;
Init();
removable=Judge(str,expstart,expend);
if(removable)
BT=CreateBT(str,expstart+1,expend-1);
else
BT=CreateBT(str,expstart,expend);
cout<<"\n\n先序遍历表达式二叉树得到表达式的前缀表示: ";
PreOrderBT(BT); //先序遍历
cout<<"\n\n中序遍历表达式二叉树得到表达式的中缀表示: ";
InOrderBT(BT); //中序遍历
cout<<"\n\n后序遍历表达式二叉树得到表达式的后缀表示: ";
PostOrderBT(BT); //后序遍历
cout<<"\n\n";
cout<<"\n表达式运算结果: ";
result=CalulateValue(BT); //计算表达式的结果
cout<<expression<<"="<<result;
cout<<"\n\n需要继续求值吗(y|Y or n|N)?";
c=getchar();
getchar();
}
while(c=='y'||c=='Y');
return 0;
}
