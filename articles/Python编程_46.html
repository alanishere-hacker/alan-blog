import re
# 编译正则表达式，提高匹配效率
pattern = re.compile(
    r"cs_item_sk\s*=\s*(\d*1)\s+.*?;\s*true\s+(\d+)",
    re.IGNORECASE
)
# 读取日志文件并逐行匹配
with open('log.txt', 'r', encoding='utf-8') as file:
    for line in file:
        match = pattern.search(line)
        if match:
            cs_item_sk_value = match.group(1)
            number_after_true = match.group(2)
            print((cs_item_sk_value, number_after_true))

import re
# 正确密码的验证函数
def is_valid_password(password):
    # 1. 检查是否包含空白字符（空格、制表符、换行等）
    if re.search(r"\s", password):
        return False
    # 2. 必须同时包含字母、数字、特殊字符
    has_digit = re.search(r"\d", password)
    has_alpha = re.search(r"[a-zA-Z]", password)
    has_special = re.search(r"[!@#$%^&*.?]", password)
    return bool(has_digit and has_alpha and has_special)
# 用户输入多个密码
user_input = input("请输入多个密码（使用英文逗号分隔）：\n")
# 使用 split 分割后，**不使用 strip 影响原始密码结构**
passwords = user_input.split(",")
# 输出符合条件的密码
print("\n符合要求的密码有：")
for pwd in passwords:
    original_pwd = pwd  # 保留原始输入（含空格）
    if is_valid_password(original_pwd):
        print(original_pwd)

import re
def parse_one_page(html):
    pattern = re.compile(
        r'<i class="board-index.*?">(?P<rank>\d+)</i>.*?'
        r'data-src="(?P<img>.*?)".*?'
        r'title="(?P<title>.*?)".*?</a>.*?'
        r'<p class="star">主演：(?P<star>.*?)</p>.*?'
        r'<p class="releasetime">上映时间：(?P<time>.*?)</p>.*?'
        r'<i class="integer">(?P<int>\d+\.)</i><i class="fraction">(?P<frac>\d+)</i>',
        re.S
    )
    result = pattern.findall(html)
   
    if not result:
        print("未匹配到任何内容，请检查 HTML 结构或正则表达式！")
        return
    print("匹配到的电影信息如下：\n")
    for i, item in enumerate(result, start=1):
        rank, img, title, star, time, integer, fraction = item
        print(f"第 {i} 部电影：")
        print(f"排名       ：{rank}")
        print(f"图片地址   ：{img}")
        print(f"电影名     ：{title}")
        print(f"主演       ：{star.strip()}")
        print(f"上映时间   ：{time.strip()}")
        print(f"评分       ：{integer + fraction}")
        print("-" * 40)
# 读取网页源码
try:
    with open('film.html', 'r', encoding='utf-8') as f:
        html_content = f.read()
        print("文件读取成功，正在解析...\n")
        parse_one_page(html_content)
except FileNotFoundError:
    print("未找到 film.html 文件，请确保它在当前目录下！")
print("xxx")
print("xxxxxxxxxxxxx")
