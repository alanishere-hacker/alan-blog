Python多线程TCP全连接端口扫描器开发

一、实验目的及要求
1. 掌握 Python3 编写脚本实现 TCP 端口扫描
实验原理与内容
在渗透测试的初步阶段通常都需要对攻击目标进行信息搜集，而端口扫描就是信息搜集中至关重要的一个步骤。通过端口扫描可以了解到目标主机都开放了哪些服务，甚至能根据服务猜测可能存在某些漏洞。 TCP 端口扫描一般分为以下几种类型：
TCP connect 扫描：也称为全连接扫描，这种方式直接连接到目标端口，完成了 TCP三次握手的过程，这种方式扫描结果比较准确，但速度比较慢而且可轻易被目标系统检测到。
TCP SYN 扫描：也称为半开放扫描，这种方式将发送一个 SYN 包，启动一个 TCP 会话，并等待目标响应数据包。如果收到的是一个 RST 包，则表明端口是关闭的，而如果收到的是一个 SYN/ACK 包，则表示相应的端口是打开的。
Tcp FIN 扫描：这种方式发送一个表示拆除一个活动的 TCP 连接的 FIN 包，让对方关闭连接。如果收到了一个 RST 包，则表明相应的端口是关闭的。
TCP XMAS 扫描：这种方式通过发送 PSH、FIN、URG、和 TCP 标志位被设为 1 的数据包。如果收到了一个 RST 包，则表明相应的端口是关闭的。
全连接扫描方式的核心就是针对不同端口进行 TCP 连接，根据是否连接成功来判断端口是否打开，现在要求采用全连接方式实现一个多线程的端口扫描器。

#!/usr/bin/env python3
import socket
import argparse
import concurrent.futures
import sys
# 扫描单个端口的函数，采用 TCP 全连接扫描方式
def scan_port(ip, port, timeout=1):
    try:
        # 创建 socket 对象并设置超时时间
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(timeout)
        # 尝试连接目标IP与端口
        result = sock.connect_ex((ip, port))
        sock.close()
        # 返回扫描结果：0 表示端口开放
        if result == 0:
            return port, True
        else:
            return port, False
    except Exception as e:
        return port, False
# 针对单个 IP 进行端口扫描
def scan_ip(ip, port_range):
    print(f"\n正在扫描 {ip} ...")
    open_ports = []
    # 使用 ThreadPoolExecutor 实现多线程扫描
    with concurrent.futures.ThreadPoolExecutor(max_workers=100) as executor:
        # 构造所有端口的扫描任务
        future_to_port = {executor.submit(scan_port, ip, port): port for port in port_range}
        for future in concurrent.futures.as_completed(future_to_port):
            port = future_to_port[future]
            try:
                port, is_open = future.result()
                if is_open:
                    print(f"端口 {port} 开放")
                    open_ports.append(port)
            except Exception as exc:
                print(f"扫描端口 {port} 时发生异常: {exc}")
    if not open_ports:
        print("没有发现开放的端口")
    return open_ports
# 解析命令行参数
def parse_args():
    parser = argparse.ArgumentParser(
        description="多线程 TCP 全连接端口扫描器"
    )
    parser.add_argument("ips", nargs="+", help="要扫描的目标IP地址列表")
    parser.add_argument(
        "-p", "--ports",
        default="1-1024",
        help="扫描端口范围, 格式为 start-end (默认: 1-1024)"
    )
    parser.add_argument(
        "-t", "--timeout",
        type=float,
        default=1.0,
        help="socket连接超时时间，单位秒 (默认: 1.0秒)"
    )
    return parser.parse_args()
# 将端口范围字符串解析为整数列表
def parse_port_range(port_range_str):
    try:
        start, end = port_range_str.split("-")
        start = int(start)
        end = int(end)
        if start > end or start < 1 or end > 65535:
            raise ValueError("端口范围错误：请确保 1<=start<=end<=65535")
        return range(start, end + 1)
    except Exception as e:
        print(f"解析端口范围失败: {e}")
        sys.exit(1)
def main():
    args = parse_args()
    port_range = parse_port_range(args.ports)
   
    # 针对每个 IP 地址进行扫描
    for ip in args.ips:
        scan_ip(ip, port_range)
if __name__ == '__main__':
    main()
