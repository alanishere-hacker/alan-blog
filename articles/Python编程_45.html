实验原理与内容
套接字(Socket) 是实现网络编程进行数据传输的一种技术手段,网络上各种各样的网络服务大多都是基于Socket 来完成通信的。socket是传输层提供给应用层的编程接口。所以，套接字socket编程分为TCP与UDP两类。在python中，通过Python套接字编程模块：import socket提供socket。通信时服务端需要建立一个socket，客户端也需要一个socket。
Socket是面向C/S（客户端/服务器）模型设计，客户端在本地随机申请一个唯一的Socket号，服务器拥有公开的socket，任何客户端都可以向它发送连接请求和信息请求。Socket数据处理流程如图：
简单的TCP服务器和客户端基本步骤：
服务端：
创建套接字，绑定套接字到本地IP与端口：socket.socket() , s.bind()
开始监听连接：s.listen()
进入循环，不断接受客户端的连接请求：s.accept()
接收传来的数据，或者发送数据给对方：s.recv() , s.sendall()
传输完毕后，关闭套接字：s.close()
客户端:
创建套接字，连接服务器地址：socket.socket() , s.connect()
连接后发送数据和接收数据：s.sendall(), s.recv()
传输完毕后，关闭套接字：s.close()
TCP和UDP是OSI七层模型中传输层提供的协议，提供可靠端到端的传输服务。
TCP（Transmission Control Protocol，传输控制协议），面向连接协议，双方先建立可靠的连接，再发送数据。适用于可靠性要求高的应用场景。
UDP（User Data Protocol，用户数据报协议），面向非连接协议。发送方直接发送数据包给接收方，相对于TCP（需要建立连接）而言，UDP不需要建立连接。因此，UDP的传输速度较快，但不保证数据传输的可靠性。UDP不提供可靠性保证。它只是简单地发送数据包给接收方，不进行确认和重传，如果在传输过程中有数据包丢失或出错，UDP不会进行纠正，因此可能导致数据丢失或乱序。UDP适用于对传输速度和实时性要求较高，且允许数据丢失的场景，如实时游戏、视频和音频流传输等。

import socket
import threading
#创建socket
serversocket = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
#绑定端口号
serversocket.bind(('127.0.0.1', 9001))
serversocket.listen(95)
#创建一个字典，用于存储用户
mydict = {}
#创建list，用于维护所有的客户端连接
mylist = []
def chatMsgToOthers(exceptMe,chatMsg):
    print(chatMsg)
    for c in mylist:
        if c.fileno() != exceptMe:
            try:
                c.send(chatMsg.encode())
            except:
                pass
def subThreadProcess(myconnection,connNum):
    #接受 用户名：
    username = myconnection.recv(1024).decode()
    mydict[myconnection.fileno()] = username
    #将客户端连接添加到list中
    mylist.append(myconnection)
    #向除了当前连接的客户端外，向所有的其他客户端发送消息
    chatMsgToOthers(connNum,'欢迎%s进入聊天室...'% username)
    while True:
        try:
            #接受客户端发送的消息
            recvMsg = myconnection.recv(1024).decode()  #接收客户端消息 阻塞式
            if recvMsg:
                #将消息发送给除了当前连接的客户端外的其他客户端
                chatMsgToOthers(connNum,'%s说：%s'%(mydict[connNum],recvMsg))
        except:
            chatMsgToOthers(connNum,'系统提示：%s已经离开聊天室...'% mydict[connNum])
            #如果客户端异常，则从字典中删除该用户
            del mydict[connNum]
            #从list中删除该用户
            mylist.remove(myconnection)
while True:
    #获取客户端连接

chat_client.py:
import socket
import threading
#与服务器通信tcp socket
sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
#连接服务器
try:
    sock.connect(('127.0.0.1', 9001))
except:
    print("连接服务器失败！")
# sock.connect(('127.0.0.1', 9001))
#向服务器发送标志：我来了
sock.send(b'1')
#接收服务器连接完成后 第一次 发送过来的消息
print(sock.recv(1024).decode())
#主动向服务器发送我的用户名
username = input("请输入你的用户名: ")
sock.send(username.encode())
#创建两个线程，分别发送数据和接收数据：
def sendThreadProcess():
    while True:
        #发送数据
        try:
            sendMsg = input(f'{username}:')
            sock.send(sendMsg.encode())
        except:
            print("发送失败，服务器已关闭")
            break
def recvThreadProcess():
    while True:
        #接收数据
        try:
            recvMsg = sock.recv(1024)
            if recvMsg:
                print(recvMsg.decode())
            else:
                pass
        except:
            print("Server closed this connection!")
#创建两个线程
sendThread = threading.Thread(target=sendThreadProcess)
recvThread = threading.Thread(target=recvThreadProcess)
threads = [sendThread, recvThread]
for t in threads:
    t.setDaemon(True)
    t.start()
t.join()
