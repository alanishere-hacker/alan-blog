#include<iostream>
#include<stdlib.h>
#include<string>
#include<iomanip>
#include<fstream>
#include<stdio.h>
#define OK 1
#define ERROR 0
#define MAXSIZE 100
#define STACK_INIT_SIZE 10
#define STACKINCREMENT 2
#define OVERLOW -2
using namespace std;
typedef struct SqStack
{
int *base;
int *top;
int stacksize;
}SqStack;
int InitStack(SqStack &S)
{
S.base=new int[MAXSIZE];
if(!S.base)
exit(OVERLOW);
S.top=S.base;
S.stacksize=MAXSIZE;
return OK;
}
int push(SqStack &S,int e)
{
if(S.top-S.base==S.stacksize)
return ERROR;
*(S.top++)=e;
return OK;
}
int Pop(SqStack &S,int *e)
{
if(S.base==S.top)
return ERROR;
S.top=S.top-1;
*e=*(S.top);
return OK;
}
int GetTop(SqStack &S)
{
if(S.top!=S.base)
return *(S.top-1);
}

{
SqStack S;
int a,b,choose,e,t;

cout<<"4.出栈\n";

choose=-1;
while(choose!=0)
{
cout<<"\n请选择："<<endl;
cin>>choose;

{
case 1:
if(InitStack(S)==1)
cout<<"初始化栈成功!\n\n";
else
cout<<"初始化栈失败!\n\n";
break;
case 2:
a=GetTop(S);

break;
case 3:
cout<<"输入的栈元素为：";

cout<<"元素入栈成功"<<endl;

cout<<"元素入栈失败"<<endl;
break;
case 4:

cout<<"出栈失败"<<endl;

{
S=NULL;
return OK;
}
int push(LinkStack &S,int e)
{
LinkStack p;
p=new StackNode;
p->data=e;
p->next=S;
S=p;
return OK;
}
int Pop(LinkStack &S,int *e)
{
LinkStack p;
p=new StackNode;
if(S==NULL)
return ERROR;
*e=S->data;
p=S;
S=S->next;
delete p;
return OK;
}
int GetTop(LinkStack &S)
{
if(S!=NULL)
return S->data;
}

顺序队列
#include<iostream>
#include<stdlib.h>
#include<string>
#include<iomanip>
#include<fstream>
#include<stdio.h>
#define OK 1
#define ERROR 0
#define OVERLOW -2
#define MAXQSIZE 100
using namespace std;
typedef struct
{
int *base;
int front;
int rear;
}SqQueue;
int InitQueue(SqQueue &Q)
{
Q.base=new int[MAXQSIZE];
if(!Q.base)
exit(OVERLOW);
Q.front=Q.rear=0;
return OK;
}
int QueueLength(SqQueue Q)
{
return (Q.rear-Q.front+MAXQSIZE)%MAXQSIZE;
}
int EnQueue(SqQueue &Q,int e)
{
if((Q.rear+1)%MAXQSIZE==Q.front)
return ERROR;
Q.base[Q.rear]=e;
Q.rear=(Q.rear+1)%MAXQSIZE;
return OK;
}
int DeQueue(SqQueue &Q,int *e)
{
if(Q.front==Q.rear)
return ERROR;
*e=Q.base[Q.front];
Q.front=(Q.front+1)%MAXQSIZE;
return OK;
}
int GetHead(SqQueue &Q)
{
if(Q.front!=Q.rear)
return Q.base[Q.front];
}
int main()
{
SqQueue Q;

链队
#include<iostream>
#include<stdlib.h>
#include<string>
#include<iomanip>
#include<fstream>
#include<stdio.h>
#define OK 1
#define ERROR 0
#define OVERLOW -2
using namespace std;
typedef struct QNode
{
int data;
struct QNode *next;
}QNode,*QueuePtr;
typedef struct
{
QueuePtr front;
QueuePtr rear;
}LinkQueue;
int InitQueue(LinkQueue &Q)
{
Q.front=Q.rear=new QNode;
Q.front->next=NULL;
return OK;
}
int EnQueue(LinkQueue &Q,int e)
{
QueuePtr p;
p=new QNode;
p->data=e;
p->next=NULL;
Q.rear->next=p;
Q.rear=p;
return OK;
}
int DeQueue(LinkQueue &Q,int *e)
{
QueuePtr p;
if(Q.front==Q.rear)
return ERROR;
p=Q.front->next;
*e=p->data;
Q.front->next=p->next;
if(Q.rear==p)
Q.rear=Q.front;
delete p;
return OK;
}
int GetHead(LinkQueue &Q)
{
if(Q.front!=Q.rear)
return Q.front->next->data;
}
