#include <stdio.h>
#include <stdlib.h>
#include <string.h>
// 定义 PCB 结构体
typedef struct FCFS {
char PrecessName[20];
struct FCFS *next;
} FCFS;
// 进程输出函数
void FCFS_Output(FCFS *L) {
FCFS *P = L->next;
printf("\nProcess execution sequence:\n");
while (P) {
printf("%-15s", P->PrecessName);
P = P->next;
}
printf("\n");
}
// 进程输入函数
void FCFS_InputProcess(FCFS *L) {
FCFS *Follow = L;
FCFS *P;
int i = 1;
char PrecessName[20] = "";
printf("Please input the process name and type 'end' to exit:\n\n");
while (1) {
printf("Process %d name: ", i);
scanf("%s", PrecessName);
if (strcmp(PrecessName, "end") == 0) {
break;
}
P = (FCFS *)malloc(sizeof(FCFS));
strcpy(P->PrecessName, PrecessName);
P->next = NULL;
Follow->next = P;
Follow = P;
i++;
}
// 如果有输入进程，则输出进程序列

}

六、实验结论与体会
（1）FCFS的特点是什么？它有什么缺点？
FCFS特点：
简单易实现：算法简单，易于实现，不需要复杂的数据结构。
公平性：按照进程到达的先后顺序调度，所有进程都会被处理。
非抢占性：一旦 CPU 开始执行某个进程，不会被其他进程打断，直到该进程完成。
吞吐量稳定：适合处理长时间的批处理任务。
FCFS缺点：
平均等待时间长：
如果一个较长的进程在短进程之前到达，会导致短进程长时间等待。
例如：如果一个需要 30 秒的进程在一个需要 2 秒的进程前面，整个队列的平均等待时间会显著增加。
低响应性：
由于等待时间不可预测，用户体验较差，特别是对于交互式系统。
不适合实时系统，因为无法保证响应时间。
资源利用率低：
可能导致资源闲置或不均衡使用，因为长任务会独占 CPU。
无法实现优先级调度：
所有进程都被视为同等重要，无法根据任务的重要性进行优先处理。
实验结论与体会：
通过本次实验，我深入理解了 FCFS调度算法的基本原理和实现过程。该算法结构简单、易于实现，但存在 “长任务阻塞短任务” 的缺点，容易引发 “拥塞效应”。在实际应用中，应根据系统需求选择更具效率和公平性的调度算法，例如 SJF 或 RR。实验过程中，我进一步掌握了链表结构的使用和内存管理的基本方法，为后续学习打下了良好的基础。
