顺序表与链表：线性表两种存储结构及基本操作实现

一、实验目的及要求
实验目的：
(1)掌握顺序表的存储结构。
(2)验证顺序表及其基本操作的实现。
(3)理解算法与程序的关系，能够将顺序表算法转换为对应的程序。
（4）掌握线性表的链式存储结构。
（5）验证链表基本操作的实现。
（6）学会使用链表来解决各种实际问题。
实验要求：
(1) 程序要添加适当的注释，程序的书写要采用缩进格式。
(2) 程序要具在一定的健壮性，即当输入数据非法时，程序也能适当地做出反应，如插入删除时指定的位置不对等等。
(3) 程序要做到界面友好，在程序运行时用户可以根据相应的提示信息进行操作。
(4) 根据实验报告模板详细书写实验报告。

实验内容：
(1）初始化顺序表。
(2）在顺序表的第 i 个位置插入元素。
(3）删除顺序表的第 i 个元素。
(4）输出顺序表。
(5）判断顺序表是否为空。
(6）判断顺序表是否为满。
(7）求顺序表第 i 个元素的值。
(8）查找值为 x 的元素。

（15）求链表中第i个元素的值。

#include<cstdio>
#include<stdlib.h>
#include<iostream>
using namespace std;
#define MAXSIZE 1000
typedef struct
{
int *elem;
int length;
}SeqList;
void InitList(SeqList &L)
{
L.elem=new int[MAXSIZE];
if(!L.elem) exit (0);
L.length=0;
}
int Creat(SeqList &L,int e)
{
for(int i=0;i<e;i++)
{
cin>>L.elem[i];
++L.length;
}
}
int Delete(SeqList &L,int e)
{
for(int j=e;j<=L.length-1;j++)
L.elem[j-1]=L.elem[j];
--L.length;
return 1;
}
int Print(SeqList &L)
{
int i;
for(i=0;i<L.length;i++)
{
printf("%d ",L.elem[i]);
}
}
int lengthList(SeqList &L)
{
return L.length;
}
int SearchList(SeqList &L,int e)
{
int i=0;
while(i<=L.length&&L.elem[i-1]!=e)
i++;
if(L.elem[i-1]==e)
return i;
else
return-1;
}
int Insert(SeqList &L,int i,int e)
{
if((i<1)||(i>L.length))
return 0;
if(L.length==MAXSIZE)
return 0;
for(int j=L.length-1;j>=i-1;j--)
L.elem[j+1]=L.elem[j];
L.elem[i-1]=e;
++L.length;
return 1;
}
int GetList(SeqList &L,int i)
{
if(i<1||i>L.length+1)
return 0;
int e=L.elem[i-1];
return 1;
}

case 2:

#include<iostream>
#include<stdlib.h>
#include<string>
#include<iomanip>
#include<fstream>
using namespace std;
#define OK 1
#define ERROR 0
#define OVERFLOW -2
#define NULL 0
typedef int Status;
typedef int dataType;
typedef struct LNode
{
dataType data;
struct LNode *next;
} LNode, *LinkList;
int length;
Status InitList_L(LinkList &L)
{
L = new LNode;
L->next = NULL;
return OK;
}
LNode *CreateList_R(LinkList &L,int n)
{
int a[n],i;
LinkList tail;
LNode *p;
L->next=NULL;
tail=L;
printf("请输入链表各结点的数据：");
for(i=0;i<n;i++)
{
scanf("%d",&a[i]);
p=(LNode *)malloc(sizeof(LNode));
p->data=a[i];
tail->next=p;
tail=p;
}
tail->next=NULL;
return L;
}
LNode *GetElem_L(LinkList &L,int i)
{
int j=1;
LinkList p=L->next;
while(p&&j<i)
{
j++;
p=p->next;
}
printf("第%d个元素的值是：%d\n",i,p->data);
}
int LocateElem_L(LinkList &L,int n)
{
LinkList p=L;
int cout=0;
while(p&&p->data!=n)
{
p=p->next;
++cout;
}
return cout;
if(p==NULL||p->data==n)
return n;
}
LNode *GetNode(LinkList &L,int i)
{
int j=0;
LNode *p=L;
while(p->next!=NULL&&j<i)
{
p=p->next;
j++;
}
if(i==j)
return p;
else return NULL;
}
int ListInsert_L(LinkList &L,int n,int i)
{
LinkList p,s;
p=GetNode(L,n-1);
if(p==NULL)
{
printf("未找到第%d个结点",n-1);
return 0;
}
s=(LinkList)malloc(sizeof(LNode));
s->data=i;
s->next=p->next;
p->next=s;
return 1;
}
int ListDelete_L(LinkList &L,int n)
{
LNode *p,*r;
int i;
p=GetNode(L,n-1);
if(p==NULL||p->next==NULL)
{
printf("未找到第%d个结点",i-1);
return 0;
}
r=p->next;
p->next=r->next;
free(r);
return 1;
}
int PrintList(LinkList &L)
{
LinkList p=L;
int e;
while(p->next!=NULL)
{
p=p->next;
e=p->data;
printf("%d ",e);
}
}
LNode *LengthList(LinkList &L)
{
LinkList p=L;
int cout=0;
while(p->next!=NULL)
{
p=p->next;
++cout;
}
printf("链表的表长为：%d",cout);
}
int EmptyList(LinkList &L)
{
LinkList p,q;
for(p=NULL,q=L->next;p;p=q,p=p->next)
{
free(p);
}
printf("链表已清空！");
}

{

double t;
int e;
LinkList L;
